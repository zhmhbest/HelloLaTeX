\section{Space}

\par ()             无空格
\par (\thinspace)   空格1/6em
\par (\enspace)     空格1/2em
\par (\quad)        空格1em
\par (\qquad)       空格2em
\par (\ )           空格
\par (~)            硬空格
\par ({\kern -1em})     自定义 kern -1
\par ({\kern 1em})      自定义 kern +1
\par ({\hskip -1em})    自定义 hskip -1
\par ({\hskip 1em})     自定义 hskip +1
\par (\hspace{-1em})    自定义 hspace -1
\par (\hspace{1em})     自定义 hspace +1
\par (\hphantom{abc})   占位符 hphantom abc
\par (\hfill)           弹性长度（撑满整个空间）
